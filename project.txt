
CREATE TABLE CUSTOMER(
    CUST_ID INT(10) PRIMARY KEY,
    CUST_NAME VARCHAR(10), 
    ACCOUNT_TYPE VARCHAR(25),
    ADDRESS TEXT,
    PINCODE VARCHAR(10),
    EMAIL VARCHAR(25),
    PASSWORD VARCHAR(25)
);

CREATE TABLE ACCOUNT (
ACCOUNT_ID INT(3) PRIMARY KEY,
CUST_ID INT(10),
ACCOUNT_NO VARCHAR(20),
NAME VARCHAR(10),
ADDRESS VARCHAR(20),
FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER (CUST_ID));

CREATE TABLE ADMIN (
ADMIN_ID INT (10),
ADMIN_NAME VARCHAR (20),
PASSWORD CHAR (1),
PRIMARY KEY (ADMIN_ID));

CREATE TABLE BILLING (
BILL_ID INT(3) PRIMARY KEY,
CUST_ID INT(10),
ACCOUNT_ID INT(10),
PAYMENT_MODE VARCHAR(20),
PAYMENT_DATE DATE,
PAYMENT_TIME TIME,
BILL_AMOUNT VARCHAR(25),
PAID_AMOUNT VARCHAR(25),
FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER (CUST_ID),
FOREIGN KEY (ACCOUNT_ID) REFERENCES ACCOUNT (ACCOUNT_ID));

CREATE TABLE ELECTRICITYBOARD (
ELECTRICITYBOARD_ID INT (10),
ELECTRICITYBOARD VARCHAR (20),
PRIMARY KEY (ELECTRICITYBOARD_ID));

CREATE TABLE FEEDBACK (
FEEDBACK_ID INT (10),
CUST_ID INT(10),
FEEDBACK TEXT,
FEEDBACK_DATE DATE,
PRIMARY KEY (FEEDBACK_ID),
FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER (CUST_ID));

CREATE TABLE TARRIF (
TARRIF_ID INT (10) PRIMARY KEY );

CREATE TABLE INVOICE(
INVOICE_ID INT(3) PRIMARY KEY,
ELECTRICITYBOARD_ID INT (10),
ACCOUNT_ID INT(10),
READINGDATE DATE,
PRESENT_READING FLOAT(10,2),
PREVIOUS_READING FLOAT(10,2),
FIXED_CHARGE FLOAT(10,2),
ENERGY_CHARGE FLOAT(10,2),
BILL_AMOUNT VARCHAR(25),
PREVIOUS_BALANCE FLOAT(10,2),
CONCESSION FLOAT(10,2),
NET_AMOUNT VARCHAR(25),
DUE_DATE DATE,
STATUS VARCHAR(10),
FOREIGN KEY (ELECTRICITYBOARD_ID) REFERENCES ELECTRICITYBOARD (ELECTRICITYBOARD_ID),
FOREIGN KEY (ACCOUNT_ID) REFERENCES ACCOUNT (ACCOUNT_ID));

DESC CUSTOMER;
DESC ACCOUNT;
DESC ADMIN;
DESC BILLING;
DESC ELECTRICITYBOARD;
DESC FEEDBACK;
DESC TARRIF;
DESC INVOICE;
